Compiling a RedHat Linux Kernel

/etc/lilo.conf


boot=/dev/sda1
map=/boot/map
install=/boot/boot.b
prompt
timeout=50
linear
default=linux-up

image=/boot/vmlinuz-2.2.16
     label=linux-up
     initrd=/boot/initrd-2.2.16.img
     read-only
     root=/dev/sda1

image=/boot/vmlinuz-2.2.16smp
     label=linux
     initrd=/boot/initrd-2.2.16.img
     read-only
     root=/dev/sda1
